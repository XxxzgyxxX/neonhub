-- Função para toggle de configurações
function ToggleSetting(option)
    if option == 1 then
        Settings.AutoFarmLevel = not Settings.AutoFarmLevel
        print("🚜 Auto Farm Level:", Settings.AutoFarmLevel and "ATIVADO" or "DESATIVADO")
        if Settings.AutoFarmLevel then
            AutoFarmLevel()
        end
    elseif option == 2 then
        Settings.FastAttack = not Settings.FastAttack
        print("⚡ Fast Attack:", Settings.FastAttack and "ATIVADO" or "DESATIVADO")
    elseif option == 3 then
        Settings.AutoStats = not Settings.AutoStats
        print("📊 Auto Stats:", Settings.AutoStats and "ATIVADO" or "DESATIVADO")
        if Settings.AutoStats then
            AutoStats()
        end
    elseif option == 4 then
        Settings.ESPPlayer = not Settings.ESPPlayer
        print("👁️ Player ESP:", Settings.ESPPlayer and "ATIVADO" or "DESATIVADO")
        PlayerESP()
    elseif option == 5 then
        if Settings.WalkSpeed == 16 then
            Settings.WalkSpeed = 100
            print("🚀 Speed Hack: ATIVADO (Velocidade: 100)")
        else
            Settings.WalkSpeed = 16
            print("🚀 Speed Hack: DESATIVADO (Velocidade: 16)")
        end
    elseif option == 6 then
        Settings.TeleportToChest = not Settings.TeleportToChest
        print("📦 Auto Chest:", Settings.TeleportToChest and "ATIVADO" or "DESATIVADO")
        if Settings.TeleportToChest then
            AutoChest()
        end
    elseif option == 7 then
        Settings.TeleportToFruit = not Settings.TeleportToFruit
        print("🍎 Auto Fruit:", Settings.TeleportToFruit and "ATIVADO" or "DESATIVADO")
        if Settings.TeleportToFruit then
            AutoFruit()
        end
    elseif option == 8 then
        Settings.AutoRejoin = not Settings.AutoRejoin
        print("🔄 Auto Rejoin:", Settings.AutoRejoin and "ATIVADO" or "DESATIVADO")
    elseif option == 9 then
        Settings.FpsBooster = not Settings.FpsBooster
        print("⚙️ FPS Booster:", Settings.FpsBooster and "ATIVADO" or "DESATIVADO")
        if Settings.FpsBooster then
            print("🎨 Qualidade gráfica reduzida para melhor performance")
        else
            print("🎨 Qualidade gráfica restaurada")
        end
    end
end
