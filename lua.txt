-- Anti-Detecção Avançada
local function setupAntiDetection()
    pcall(function()
        local mt = getrawmetatable(game)
        local old = mt.__namecall
        setreadonly(mt, false)
        mt.__namecall = newcclosure(function(self, ...)
            local method = getnamecallmethod()
            if method == "FireServer" or method == "InvokeServer" then
                if tostring(self):find("LogService") or tostring(self):find("Kick") then
                    return
                end
            end
            return old(self, ...)
        end)
        setreadonly(mt, true)
    end)
end

setupAntiDetection()

local CombatFramework, CombatFrameworkR, RigController, RigControllerR

    CombatFramework = require(Player.PlayerScripts:WaitForChild("CombatFramework"))
    CombatFrameworkR = getupvalues(CombatFramework)[2]

function FastAttack()
    if Settings.FastAttack then
        if tick() - cooldownfastattack > 0.1 then
            pcall(function()
                for i, v in pairs(CombatFrameworkR.activeController.data) do
                    if v.active then
                        if v.timeToNextAttack > 0 then
                            v.timeToNextAttack = 0
                        end
                        if v.hitboxMagnitude ~= 55 then
                            v.hitboxMagnitude = 55
                        end
                    end
                end
            end)
            cooldownfastattack = tick()
        end
    end
end

-- Tween Function
function topos(Pos)
    local Distance = (Pos.Position - Player.Character.HumanoidRootPart.Position).Magnitude
    local Speed = Distance < 25 and 10000 or Distance < 50 and 2000 or Distance < 150 and 800 or Distance < 300 and 600 or Distance < 500 and 400 or Distance < 750 and 250 or 200
    
    TweenService:Create(
        Player.Character.HumanoidRootPart,
        TweenInfo.new(Distance/Speed, Enum.EasingStyle.Linear),
        {CFrame = Pos}
    ):Play()
end

-- Auto Farm Level
function AutoFarmLevel()
    spawn(function()
        while Settings.AutoFarmLevel do
            wait()
            pcall(function()
                GetQuest()
                
                if not Player.PlayerGui.Main.Quest.Visible or not string.find(Player.PlayerGui.Main.Quest.Container.QuestTitle.Title.Text, NameMon) then
                    game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("StartQuest", NaemQuest, LevelQuest)
                end
                
                for _, v in pairs(Workspace.Enemies:GetChildren()) do
                    if v.Name == Ms and v:FindFirstChild("Humanoid") and v:FindFirstChild("HumanoidRootPart") and v.Humanoid.Health > 0 then
                        repeat
                            wait()
                            if Settings.BringMobs then
                                v.HumanoidRootPart.CanCollide = false
                                v.HumanoidRootPart.Size = Vector3.new(50,50,50)
                                v.HumanoidRootPart.CFrame = Player.Character.HumanoidRootPart.CFrame * CFrame.new(0, 0, -3)
                            end
                            topos(v.HumanoidRootPart.CFrame * CFrame.new(0, Settings.FarmDistance, 0))
                            sethiddenproperty(Player, "SimulationRadius", math.huge)
                        until not Settings.AutoFarmLevel or v.Humanoid.Health <= 0 or not v.Parent
                    end
                end
            end)
        end
    end)
end

-- Player ESP
function PlayerESP()
    if Settings.ESPPlayer then
        for _, v in pairs(Players:GetPlayers()) do
            if v ~= Player and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
                if not v.Character.HumanoidRootPart:FindFirstChild("PlayerESP") then
                    CreateESP(v.Character.HumanoidRootPart, v.Name, Color3.fromRGB(255, 255, 255))
                end
            end
        end
    end
end
