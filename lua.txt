-- Utility Functions
local function GetPlayerLevel()
    if Player.Data and Player.Data.Level then
        return Player.Data.Level.Value
    end
    return 1
end

local function GetCurrentSea()
    local placeId = game.PlaceId
    for sea, data in pairs(SeaData) do
        if data.PlaceId == placeId then
            return sea
        end
    end
    return 1
end

local function GetCurrentQuest()
    local level = GetPlayerLevel()
    local currentSea = GetCurrentSea()
    local quests = SeaData[currentSea].Quests
    
    local bestQuest = nil
    for questLevel, questData in pairs(quests) do
        if level >= questLevel then
            if not bestQuest or questLevel > bestQuest.Level then
                bestQuest = questData
                bestQuest.Level = questLevel
            end
        end
    end
    
    return bestQuest
end

local function TeleportTo(targetCFrame)
    if not HumanoidRootPart or not targetCFrame then return end
    
    local distance = (HumanoidRootPart.Position - targetCFrame.Position).Magnitude
    local speed = math.max(distance / NeonHub.TeleportSpeed, 0.1)
    
    local tweenInfo = TweenInfo.new(
        speed,
        Enum.EasingStyle.Linear,
        Enum.EasingDirection.InOut,
        0,
        false,
        0
    )
    
    local tween = TweenService:Create(HumanoidRootPart, tweenInfo, {CFrame = targetCFrame})
    tween:Play()
    
    return tween
end

local function GetNearestEnemy(enemyName)
    local nearestEnemy = nil
    local nearestDistance = math.huge
    
    if Workspace:FindFirstChild("Enemies") then
        for _, enemy in pairs(Workspace.Enemies:GetChildren()) do
            if enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") then
                if enemy.Humanoid.Health > 0 then
                    local distance = (HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
                    if distance < nearestDistance then
                        if not enemyName or enemy.Name == enemyName then
                            nearestDistance = distance
                            nearestEnemy = enemy
                        end
                    end
                end
            end
        end
    end
    
    return nearestEnemy, nearestDistance
end

local function FastAttack()
    if not NeonHub.FastAttack then return end
    
    pcall(function()
        game:GetService("ReplicatedStorage").RigControllerEvent:FireServer("weaponChange", tostring(GetCurrentWeapon()))
        game:GetService("ReplicatedStorage").RigControllerEvent:FireServer("hit", GetNearestEnemy(), 1, "")
    end)
end

local function AttackEnemy(enemy)
    if not enemy or not enemy:FindFirstChild("HumanoidRootPart") then return end
    
    pcall(function()
        -- Teleport close to enemy
        local enemyPosition = enemy.HumanoidRootPart.CFrame * CFrame.new(0, 0, NeonHub.AttackDistance)
        TeleportTo(enemyPosition)
        
        -- Fast attack
        FastAttack()
        
        -- Traditional attack
        local args = {
            [1] = "weaponUnequip",
            [2] = game.Players.LocalPlayer.Character:FindFirstChild("Combat")
        }
        
        if game:GetService("ReplicatedStorage"):FindFirstChild("Remotes") then
            local commE = game:GetService("ReplicatedStorage").Remotes:FindFirstChild("CommE")
            if commE then
                commE:FireServer(unpack(args))
            end
        end
    end)
end

local function GetQuest()
    if not NeonHub.AutoQuest then return end
    
    local questData = GetCurrentQuest()
    if not questData then return end
    
    pcall(function()
        local args = {
            [1] = "StartQuest",
            [2] = questData.Name,
            [3] = 1
        }
        
        if game:GetService("ReplicatedStorage"):FindFirstChild("Remotes") then
            local commF = game:GetService("ReplicatedStorage").Remotes:FindFirstChild("CommF_")
            if commF then
                commF:InvokeServer(unpack(args))
            end
        end
    end)
end

local function AutoFarm()
    if not NeonHub.AutoFarm then return end
    
    local questData = GetCurrentQuest()
    if not questData then return end
    
    local enemy, distance = GetNearestEnemy(questData.Enemy)
    
    if enemy and distance < NeonHub.FarmDistance then
        AttackEnemy(enemy)
    elseif not enemy then
        -- Try to find quest enemies
        local enemy2 = GetNearestEnemy()
        if enemy2 then
            AttackEnemy(enemy2)
        end
    end
end

local function AutoChest()
    if not NeonHub.AutoChest then return end
    
    pcall(function()
        for _, chest in pairs(Workspace:GetChildren()) do
            if chest.Name:find("Chest") and chest:FindFirstChild("Part") then
                local distance = (HumanoidRootPart.Position - chest.Part.Position).Magnitude
                if distance < 50 then
                    TeleportTo(chest.Part.CFrame)
                    wait(1)
                    
                    fireproximityprompt(chest.ProximityPrompt, 0)
                    wait(1)
                end
            end
        end
    end)
end

local function AutoSaber()
    if not NeonHub.AutoSaber then return end
    
    local level = GetPlayerLevel()
    local currentSea = GetCurrentSea()
    
    if level >= 200 and currentSea == 1 then
        pcall(function()
            local args = {
                [1] = "ProQuestProgress",
                [2] = "SaberQuest"
            }
            
            if game:GetService("ReplicatedStorage"):FindFirstChild("Remotes") then
                local commF = game:GetService("ReplicatedStorage").Remotes:FindFirstChild("CommF_")
                if commF then
                    commF:InvokeServer(unpack(args))
                end
            end
        end)
    end
end

local function CheckSeaProgression()
    local level = GetPlayerLevel()
    local currentSea = GetCurrentSea()
    
    pcall(function()
        if currentSea == 1 and level >= 700 then
            -- Go to Second Sea
            TeleportService:Teleport(4442272183, Player)
        elseif currentSea == 2 and level >= 1500 then
            -- Go to Third Sea  
            TeleportService:Teleport(7449423635, Player)
        end
    end)
end
