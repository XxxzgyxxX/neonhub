local function CreateGlow(parent, color, size)
    local glow = Instance.new("ImageLabel")
    glow.Name = "Glow"
    glow.Parent = parent
    glow.BackgroundTransparency = 1
    glow.Size = UDim2.new(1, size, 1, size)
    glow.Position = UDim2.new(0, -size/2, 0, -size/2)
    glow.Image = "rbxasset://textures/ui/GuiImagePlaceholder.png"
    glow.ImageColor3 = color
    glow.ImageTransparency = 0.5
    glow.ZIndex = parent.ZIndex - 1
    
    -- Anima√ß√£o de brilho
    local glowTween = TweenService:Create(glow, TweenInfo.new(2, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), {
        ImageTransparency = 0.8
    })
    glowTween:Play()
    
    return glow
end
